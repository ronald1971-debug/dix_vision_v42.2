### 5.3 Enhanced Patch Generation

```python
from containers.system_core.evolution_engine.dyon import get_enhanced_patch_generator

# Get enhanced patch generator
patch_generator = get_enhanced_patch_generator()

# Generate safe patches for a target path
result = patch_generator.generate_safe_patches(
    target_path="containers/system_core/evolution_engine",
    patch_types=[
        PatchType.REFACTOR_EXTRACT_METHOD,
        PatchType.REFACTOR_DECOMPOSE_CONDITIONAL,
        PatchType.FIX_CODE_SMELL
    ]
)

print(f"Generated patches: {len(result.generated_patches)}")
print(f"Safe patches: {len(result.safe_patches)}")
print(f"Conditionally safe: {len(result.conditionally_safe_patches)}")
print(f"Unsafe: {len(result.unsafe_patches)}")

# Apply safe patches
for patch in result.safe_patches:
    success = patch_generator.apply_safe_patch(patch)
    if success:
        print(f"Applied patch: {patch.patch_id}")
```
